public class FixBuzz
{
    /**
     * Print Fix Buzz
     *
     * @param div
     * @return
     */
    public String divideByKata(int div) {
        String rest;

        if (div % 3 == 0 && div % 5 != 0 ) {
            rest = "Fizz";
        } else if (div % 5 == 0 && div % 3 != 0) {
            rest = "Buzz";
        } else if (div % 3 == 0 && div % 5 == 0) {
            rest = "Fizz Buzz";
        } else
            rest = String.valueOf(div);

        return rest;
    }
